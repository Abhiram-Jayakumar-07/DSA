char stack[100]
int top = -1

function push(int x):
    top = top + 1
    stack[top] = x

function pop():
    if top == -1:
        return '#'

    int x = stack[top]
    top = top - 1
    return x

function check_brackets(char expr[], int len):
    for i=0 to len-1:
        if expr[i] == '(' OR expr[i] == '[' OR expr[i] == '{':
            push(expr[i])
        else if expr[i] == ')' AND pop() != '(':
            return 0
        else if expr[i] == ']' AND pop() != '[':
            return 0
        else if expr[i] == '}' AND pop() != '{':
            return 0
    if top != -1:
        return 0
    return 1
